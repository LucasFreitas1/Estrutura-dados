\section{Grafos}
\begin{frame}{Grafo}
	\begin{itemize}
		\item Grafos são estruturas de dados parecidas com árvores.
		\item Em um sentido matemático, uma árvore é um tipo de grafo.
		\item Grafos são flexíveis e excelentes para representação de problemas físicos ou abstratos.
		\item Por exemplo, os nós em um Grafo podem representar cidades e as arestas podem representar rotas de vôo de linhas aéreas entre duas cidades.
		\item Grafo tem ampla utilização na solução de problemas relacionados à área da ciência da computação.
		\item Alguns problemas que podem ser tratados por meio de grafos são:	
			\begin{enumerate}
				\item Determinação de rotas de mensagem e/ou tráfego em geral.
				\item Modelagem de circuitos digitais.
				\item Representação de processos em um sistema paralelo ou distribuído.
			\end{enumerate}
	\end{itemize}
\end{frame}

\begin{frame}{Grafo}
	\begin{block}{Definição}
	  Um grafo é um conjunto de \alert{nós ou vértices (V)} e um \alert{conjunto de arestas ou arcos} (A).
	\end{block}
	
	\begin{block}{Definição}
	  Um grafo \alert{\textbf{não-direcionado}} é um conjunto de pontos com linhas conectando alguns dos pontos. Os pontos são conhecidos com \alert{nós} ou \alert{vértices}, e as linhas são chamadas \alert{arestas}.
	\end{block}	
\end{frame}

\begin{frame}{Grafo}
\begin{block}{Grafo direcionado}
  Um grafo direcionado $G$ é uma dupla $(V,A)$, que se pode definir a ordem nos pares de nós onde:  
		\begin{enumerate}
			\item $V$: representa um conjunto finito de nós.
			\item $A$: representa a relação binária que pode existir entre os nós.
		\end{enumerate}
\end{block}

\end{frame}

\begin{frame}{Grau de um nó}
\begin{itemize}
	\item A quantidade de arcos incidentes em um nó é chamado de grau do nó.
	\item O grau de um nó é definido como sendo o grau de saída + grau de entrada.
\end{itemize}
\begin{columns}
\column{5cm}
\begin{figure}[ht]
	\centering
	\includegraphics[width=.5\textwidth]{Exemplo-Grau-No}
	\label{fig:grau-no}
\end{figure}
\column{5cm}
\begin{itemize}
	\item O nó A tem grau igual a 2 (1 saída + 1 entrada).
	\item O nó b tem grau igual a 5 (3 saídas + 2 entradas).
\end{itemize}
\end{columns}
\end{frame}


\begin{frame}{Grafo cíclico}
\begin{itemize}
	\item Um grafo é considerado \alert{\textbf{cíclico}} quando existir um caminho de nó mínimo três nós que começam e terminam no mesmo nó.
	\item Um ciclo ocorre quando um arco começa e termina no mesmo nó.
\end{itemize}

\begin{figure}[ht]
	\centering
	\includegraphics{Exemplo-Grau-Ciclico}

\end{figure}
\end{frame}

\begin{frame}{Grafo Acíclico}
\begin{itemize}
	\item Um grafo é considerado \alert{\textbf{acíclico}} quando não ocorre nenhum ciclo em circuitos nos nós, ou seja, a orientação das setas não termina no mesmo nó.
\end{itemize}

\begin{frame}{Estrutura para armazenar Grafo}
\begin{itemize}
	\item Para representar um grafo são necessários dois conjuntos:	
		\begin{enumerate}
			\item Um para conter os nós ou vértices do grafo.
			\item Um conjunto para conter os arcos ou arestas.
		\end{enumerate}
		\item Esses dois conjuntos podem ser representados pelas seguintes estruturas de dados:		
		\begin{enumerate}
			\item Lista de adjacência.
			\item Matriz de adjacência.
		\end{enumerate}
\end{itemize}
\end{frame}

\begin{frame}{Lista de Adjacência}
\begin{itemize}
	\item A lista de adjacência para um Grafo $G$ usa um vetor com $n$ listas ligadas.
	\item Cada posição do vetor corresponde a um Nó de $G(V,A)$.
	\item Os arcos de um nó para outros são representados por listas ligadas.
\end{itemize}

\begin{figure}[ht]
	\centering
	\includegraphics[width=1.6in]{Lista-adjacencia}
	\caption{Representação de grafo através de lista de adjacência}
	\label{fig:lista-adjacencia}
\end{figure}
\end{frame}

\begin{frame}{Matriz de Adjacência}
\begin{itemize}
	\item A matriz de adjacência para um grafo $G$ assume que os nós são enumerados de 1 até $N$.
	\item A matriz de adjacência com dimensão $NxN$ de elementos, que representam os vértices.
\end{itemize}

\begin{frame}{Aplicações de Grafos}
\begin{itemize}
	\item Os grafos podem ser úteis como auxílio na determinação de caminho máximo ou mínimo entre cidades.
	\item Cidades podem ser representadas como vértices e as distâncias entre as cidades representadas como as arestas entre os vértices.
	\item Fluxo máximo (veículos, aeronaves, dados, corrente elétrica, etc.) em uma rede de fluxo.
	\item Caminho Euleriano.
\end{itemize}
\end{frame}

\begin{frame}{Caminho Euleriano}.
\begin{enumerate}
	\item Uma aplicação interessante que pode ilustrar o uso de grafos na solução de problemas rotineiros é o chamado problema do caminho \textit{Euleriano}.
	\item Nesse contexto temos o problema do carteiro chinês.
	\item O problema do carteiro chinês é determinar como passar pelas ruas uma única vez, representadas por arestas em um grafo, como forma de caminhamento.
	\item Um grafo admite um caminho \textit{Euleriano} se, e somente se, o número de arcos em cada vértice for par.	
\end{enumerate}
\end{frame}

\begin{frame}{Inspetor de estradas}
\begin{block}{Inspetor de estradas}
Um funcionário encarregado de verificar o estado das estradas, deseja planejar a sua rota de inspeção. Idealmente, esta rota deveria iniciar na capital e percorrer cada estrada exatamente uma vez, retornando ao ponto de partida. Existe tal rota?
\end{block}

\begin{itemize}
	\item Esse problema foi estudado pela primeira vez por Leonhard Euler(1707-1783) e ficou imortalizada como o
\textit{Problema das Pontes de Königsberg}.
\end{itemize}
\end{frame}

	\item O objetivo é percorrer exatamente uma vez todas as sete pontes da cidade que conectam as duas ilhas

\begin{frame}{Problema das Pontes de Königsberg}
\begin{itemize}
	\item Euler mostrou a não existência de tal circuito através de um argumento simples. Considere, por exemplo, a ilha da direita (\textit{d}). 
	\item Um caminho qualquer deve chegar à ilha e sair dela o mesmo número de vezes. 
	\item Logo, para que exista um caminho \textit{Euleriano}, deve haver um número par de pontes com extremidade nesta ilha.
\end{itemize}
\end{frame}

\begin{frame}{Problema do caminho mais curto}
\begin{itemize}
	\item O problema do caminho mais curto consiste em determinar qual o melhor caminho para atravessar uma rede saindo de um ponto (origem) e chegando em outro (destino) com o menor custo possível.
\end{itemize}

\begin{figure}[ht]
	\centering

\begin{itemize}
	\item Qual(is) o(s) menor(es) caminho(s) para atravessar o grafo saído do vértice 1 e chegando ao vértice 2.
\end{itemize}
\end{frame}
