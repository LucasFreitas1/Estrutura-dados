\Question Escreva uma função recursiva que receba dois números inteiros positivos \textit{k} e \textit{n} e calcule o valor de $k^n$. 

\parbox[c]{15.3cm}
{
  \vspace{0.5cm}
  \textbf{Solução}\\
}
  
  
\begin{lstlisting}
int potencia(int k, int n)
{
   if (n == 0)
   {
     return 1;
   }else
   {
     n = n - 1;
     return k * potencia(k, n);  
    } 
}
\end{lstlisting}

\Question A seguinte função calcula o Máximo Divisor Comum (MDC) dos inteiros positivos \textit{m} e \textit{n}. Escreva uma função recursiva equivalente.

\begin{lstlisting}
int mdc(int m, int n){
 int r;
 do{
   r = m % n;
   m = n; n = r;
 }while(r != 0); 
 return m;
}
\end{lstlisting}

\Question Escreva uma função que converta um número inteiro de decimal para binário.

\parbox[c]{15.3cm}
{
  \vspace{0.5cm}
  \textbf{Solução}\\
}
\begin{lstlisting}
/**
 * Apresenta a representação binária do número n em decimal.*/
void decimal2Binario(int n){
  if (n == 0 || n == 1)
       print(n);
  else{
    decimal2Binario(n/2);
    print(n % 2);        
  }  
}
\end{lstlisting}

\Question Usando recursão, calcule a soma de todos os valores de um \textit{array}.

\parbox[c]{15.3cm}
{
  \vspace{0.5cm}
  \textbf{Solução}\\
}
\begin{lstlisting}
/**
 * Realiza a soma de todos os elementos do vetor v de tamanho n.
 */
int soma(int v[], int n)
{
  if (n < 0)
  {
    return 0;
  } else 
  {
	return v[n - 1] + soma(v, n-- - 1);   
  }
}
\end{lstlisting}
