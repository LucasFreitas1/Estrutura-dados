\begin{lstlisting}[label=pilha,caption=Implementação de Pilha em Java]
package br.projecao.ed.linear;

/**
 * Implementação da estrutura de uma Pilha estática sequencial linear. 
 * Uma {@link Pilha} é uma um conjunto ordenado de itens no qual novos itens podem ser inseridos e a partir do qual podem ser eliminados em uma extremidade denominada topo da pilha. 
 * Por ser uma implementação estática a {@link Pilha} tem um tamanho fixo, o que significa que depois de instanciada a capacidade da pilha não muda.
 * 
 * @author alessandro.leite
 */
public class Pilha
{

	/**
	 * Elementos da {@link Pilha}.
	 */
	private Object[] elementos;

	/**
	 * Determina o topo da {@link Pilha}
	 */
	private Integer topo;

	/**
	 * Cria uma pilha vazia e com um dado tamanho.
	 * 
	 * @param tamanho
	 *            Capacidade da {@link Pilha} a ser criada. Deve ser um valor maior ou igual a zero.
	 */
	public Pilha(int tamanho)
	{
		this.elementos = new Object[tamanho];
		this.topo = 0;
	}

	/**
	 * Empilha um dado elemento se a pilha não esteja cheia.
	 * 
	 * @param elemento
	 *            Elemento a ser empilhado.
	 * @return <code>true</code> se o elemento estiver sido empilhado, caso contrário retorna
	 *         <code>false</code>.
	 */
	public boolean push(Object elemento)
	{
		if (this.topo < this.elementos.length)
		{
			this.elementos[topo++] = elemento;
		}
		return false;
	}

	/**
	 * Desempilha e retorna o elemento que está no topo da pilha, desde que a pilha não esteja
	 * vazia.
	 * 
	 * @return O elemento que está no topo ou <code>null</code> se a pilha estiver vazia.
	 */
	public Object pop()
	{
		return isEmpty() ? null : this.elementos[topo--];
	}

	/**
	 * Retorna uma nova {@link Pilha} com os elementos na ordem inversa dessa {@link Pilha}. A nova
	 * {@link Pilha} tem tamanho igual a essa {@link Pilha}.
	 * 
	 * @return Uma nova {@link Pilha} com os elementos na ordem inversa dessa {@link Pilha}.
	 */
	public Pilha inverter()
	{
		Pilha inversa = new Pilha(this.elementos.length);
		inversa.topo = this.topo;

		for (int i = 0, j = this.topo - 1; i < this.topo; i++, j--)
		{
			inversa.elementos[i] = this.elementos[j];
		}
		return inversa;
	}

	/**
	 * Retorna uma cópia dessa {@link Pilha}.
	 * 
	 * @return Uma cópia dessa {@link Pilha}
	 */
	public Pilha copia()
	{
		Pilha copia = new Pilha(this.elementos.length);
		copia.topo = this.topo;

		for (int i = 0; i < this.topo; i++)
		{
			copia.elementos[i] = this.elementos[i];
		}
		return copia;
	}

	/**
	 * Empilha todos os elementos de uma dada {@link Pilha} a essa {@link Pilha}, sem modificar o
	 * estado da {@link Pilha} dada para ser concatenada com essa. Os elementos são empilhados até a
	 * capacidade máxima da {@link Pilha}.
	 * 
	 * @param p
	 *            Pilha que deve ter todos os seus elementos empilhados nessa {@link Pilha}.
	 */
	public void concatena(Pilha p)
	{
		// realiza uma cópia da pilha p para que ela não tenha o seu estado modificado.
		Pilha copia = p.copia();

		while (!copia.isEmpty())
		{
			this.push(copia.pop());
		}
	}

	/**
	 * Realiza uma pesquisa por um dado valor na {@link Pilha}, onde se o elemento estiver presente
	 * retorna 1 caso contrário retorna 0.
	 * 
	 * @param value
	 *            Valor a ser pesquisado na {@link Pilha}.
	 * @return O valor 1 se o elemento estiver presente, ou 0 (zero) caso contrário.
	 */
	public int search(Object value)
	{
		int i = this.topo;

		while (i >= 0 && this.elementos[i] != value)
			i--;

		return i < 0 ? 0 : 1;
	}

	/**
	 * Apresenta no console {@link System#out} os elementos da {@link Pilha}.
	 */
	public void print()
	{
		for (int i = topo - 1; i >= 0; i--)
		{
			System.out.println(this.elementos[i]);
		}
	}

	/**
	 * Retorna a quantidade de elementos presentes na pilha, ou seja, a altura da {@link Pilha}. O
	 * tamanho da pilha é diferente a sua capacidade. A capacidade representa a quantidade máxima de
	 * elementos permitidos em uma pilha, enquanto o tamanho, representa a quantidade de elementos
	 * que foram adicionados.
	 * 
	 * @return
	 */
	public int size()
	{
		return this.topo;
	}

	/**
	 * Retorna <code>true</code> se a pilha estiver vazia ou <code>false</code> contrário. Um pilha
	 * é considerada vazia quando não há nenhum elemento empilhado.
	 * 
	 * @return <code>true</code> se a pilha estiver vazia ou <code>false</code> contrário.
	 */
	public boolean isEmpty()
	{
		return this.topo == 0;
	}
}

{
	private Pilha vinhos = new Pilha(Integer.MAX_VALUE);

	/**
	 * Inclui um novo {@link Vinho} na {@link Adega}
	 * 
	 * @param vinho
	 *            {@link Vinho} a ser adicionado à {@link Adega}.

	public void incluir(Vinho vinho)

		this.vinhos.push(vinho);
	}

	 * Retorna o {@link Vinho} que deve ser aberto. O {@link Vinho} a ser aberto é sempre o último
	 * que foi adquirido, ou seja, o {@link Vinho} adquirido recentemente.
	 * 
	 * @return O {@link Vinho} que deve ser aberto.
	 */
	public Vinho abrir()
	{
		return (Vinho) this.vinhos.pop();
	}

	/**
	 * Retorna os cinco vinhos mais antigos da {@link Adega}.
	 * 
	 * @return Os cinco vinhos mais antigos da {@link Adega}.
	 */
	public Vinho[] obterCincoVinhosMaisAntigos()
	{
		Vinho[] aquisicoes = new Vinho[5];

		Pilha vinhos = this.vinhos.inverter();

		int i = 0;
		while (i < 5 && !vinhos.isEmpty())
		{
			aquisicoes[i++] = (Vinho) vinhos.pop();
		}

		return aquisicoes;
	}
}

\Question Uma pilha pode ser usada para rastrear os tipos de escopos encontrados em uma expressão matemática e verificar se o uso deles está correto. Os delimitadores de escopo podem ser os parênteses, os colchetes e as chaves. Escreva um programa que leia uma expressão matemática e verifique se os escopos estão posicionados de forma correta.


%Para esse problema deve-se percorrer a sequência de caracteres dada da esquerda para a direita. Os caracteres $($, $[$ e $\{$ serao inseridos na pilha. No caso dos caracteres $)$, $]$ e $\}$, será verificado se o caractere que está na extremidade da pilha é do mesmo tipo do que está sendo processado. No caso positivo, ele será removido e a execução continua. No caso negativo, a sequência nao está balanceada. No fim, a sequência será considerada correta se nao há mais símbolos a processar e a pilha está vazia.

\begin{lstlisting}
	/**
	 * Verifica se os escopos dos delimitadores: parênteses (), 
	 * os colchetes [] e as chaves {} estão posicionados 
	 * de forma correta.
	 * 
	 * @param expressao Expressão matemática a ser verificada se 
	 * 									os escopos dos delimitadores estão posicionados 
	 * 									de forma correta.
	 * @return <code>true</code> se os escopos dos delimitadores estão
	 *         posicionados de forma correta ou <code>false</code> caso
	 *         contrário.
	 */
	boolean balanceada(String expressao) {
		Pilha<Character> pilha = new Pilha<Character>();

		for (char c : expressao.toCharArray()) {
			switch (c) {
			case '(':
			case '[':
			case '{':
				pilha.push(c);
				break;
			case ')':
				if (pilha.isEmpty())
					return false;
				else {
					char d = pilha.pop();
					if ('(' != d)
						return false;
				}
				break;
			case ']':
				if (pilha.isEmpty())
					return false;
				else {
					char d = pilha.pop();
					if ('[' != d)
						return false;
				}
				break;
			case '}':
				if (pilha.isEmpty())
					return false;
				else {
					char d = pilha.pop();
					if ('{' != d)
						return false;
				}
				break;
			}
		}
		return pilha.isEmpty();
	}
\end{lstlisting}
