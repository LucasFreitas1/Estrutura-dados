\question[1\half] A formatação de um documentação html é definida por tags. As tags são usados em pares: um de abertura e um de fechamento. Exemplo $<em>$\textit{itálico}$</em>$  e  $<strong>$\textbf{negrito}$</strong>$. Escreva um programa que verifique se as tags de um arquivo html estão corretamente fechadas.  Considere apenas as tags strong, tt, small, big, h1, blockquote, div, table, tr, td, address, head, body, html.
\begin{solution}
\begin{lstlisting}
	/**
	 * Verifica se as tags html: strong, em, tt estão posicionados de forma
	 * correta, ou seja, se para cada tag de abertura há uma correspondente
	 * fechando segundo o seu escopo de modo a estarem balanceadas.
	 * 
	 * @param tags
	 *            Tags html a serem verificadas se estão balanceadas.
	 * @return <code>true</code> se as tags estiverem balanceadas, ou seja
	 *         posicionados de forma correta ou <code>false</code> caso
	 *         contrário.
	 */
	public boolean balanceada(String[] tags) {
		Pilha<String> pilha = new Pilha<String>();

		for (String tag : tags) {

			if (isTagAbertura(tag)) {
				pilha.push(tag);
				continue;
			}

			if (tag.equals("</strong>")) {
				if (pilha.isEmpty() || !"<strong>".equals(pilha.pop()))
					return false;
			}

			if (tag.equals("</em>")) {
				if (pilha.isEmpty() || !"<em>".equals(pilha.pop()))
					return false;
			}

			if (tag.equals("</tt>")) {
				if (pilha.isEmpty() || !"<tt>".equals(pilha.pop()))
					return false;
			}
		}
		return pilha.isEmpty();
	}

	/**
	 * Verifica se uma dada palavra é uma tag html de abertura.
	 * 
	 * @param tag
	 *            Palavra a ser verificada se trata-se de uma tag html de
	 *            abertura. Exemplo de tags: strong, em, tt.
	 * @return <code>true</code> se for uma tag html de abertura ou
	 *         <code>false</code> caso contrário.
	 */
	private boolean isTagAbertura(String tag) {
		return tag.equals("<strong>") || tag.equals("<em>")
				|| tag.equals("<tt>");
	}
\end{lstlisting}
\end{solution}

\question[1\half] Implemente um algoritmo que teste se duas pilhas são ou não iguais. O algoritmo deve retornar 1 se as duas pilhas forem iguais, e 0 (zero) caso contrário. Ao final da execução, as duas pilhas devem estar no mesmo estado em que estavam no início.
\begin{solution}
\begin{lstlisting}
	/**
	 * Verifica se duas pilhas são iguais, mudar o estado das pilhas. Duas pilhas
	 * são iguais quando possuem os mesmos elementos.
	 * 
	 * @param p1
	 *            Pilha a ser verificada se é igual a pilha P2.
	 * @param p2
	 *            Pilha a ser verificada se é igual a pilha P1.
	 * @return
	 */
	public static int iguais(Pilha<?> p1, Pilha<?> p2) {

		if (p1.size() != p2.size()) {
			return 1;
		} else {

			Pilha<?> p3 = p1.copia();
			Pilha<?> p4 = p2.copia();

			while (!p3.isEmpty() && p3.pop() == p4.pop())
				;

			return p3.isEmpty() ? 1 : 0;
		}
	}
\end{lstlisting}
\end{solution}
\question [2] Implemente as operações de \textbf{inserir} e \textbf{remover} de uma \textbf{Fila} utilizando uma \textbf{Lista Encadeada Simples}.  Compute a complexidade de tempo de cada função.

\begin{solution}
\begin{lstlisting}
public class Fila<E> {

	private ListaEncadeada<E> elementos = new ListaEncadeada<E>();

	/**
	 * Adiciona um dado elemento no final da fila em tempo constante O(1).
	 * 
	 * @param elemento
	 *            Elemento a ser adicionado.
	 */
	public void adicionar(E elemento) {
		elementos.adicionarAoFinal(elemento);
	}

	/**
	 * Remove e retorna o primeiro elemento da fila, desde que a fila não esteja
	 * vazia em tempo constante O(1).
	 * 
	 * @return O primeiro elemento da fila que foi removido, ou
	 *         <code>null</code> se a fila estiver vazia.
	 */
	public E remover() {
		return this.isEmpty() ? null : elementos.removerDoInicio();
	}

	/**
	 * Retorna <code>true</code> se essa fila estiver vazia ou
	 * <code>false</code> caso contrário.
	 * 
	 * @return <code>true</code> se essa fila estiver vazia ou
	 *         <code>false</code> caso contrário.
	 */
	public boolean isEmpty() {
		return this.elementos.isEmpty();
	}
}
\end{lstlisting}
\end{solution}
